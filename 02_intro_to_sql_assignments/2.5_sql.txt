1->
SELECT JOB,
       SUM(DECODE(DEPTNO, 10, SAL, 0)) AS "Dept 10",
       SUM(DECODE(DEPTNO, 20, SAL, 0)) AS "Dept 20",
       SUM(DECODE(DEPTNO, 30, SAL, 0)) AS "Dept 30",
       SUM(SAL) AS "Total Salary"
FROM EMP
GROUP BY JOB;


2->
SELECT TO_CHAR(DEPTNO) AS "Department/Job",
SUM(SAL) AS "Total Salary"
FROM EMP
GROUP BY DEPTNO
UNION ALL
SELECT JOB,SUM(SAL)
FROM EMP
GROUP BY JOB
UNION ALL
SELECT 'TOTAL',SUM(SAL)
FROM EMP;

3->
SELECT JOB, DEPTNO
FROM EMP
WHERE DEPTNO = 20
UNION
SELECT JOB, DEPTNO
FROM EMP
WHERE DEPTNO = 10
UNION
SELECT JOB, DEPTNO
FROM EMP
WHERE DEPTNO = 30;
