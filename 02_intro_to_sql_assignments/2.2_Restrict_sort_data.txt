1->
SELECT ENAME,SAL,COMM
FROM EMP
WHERE COMM IS NOT NULL
ORDER BY 2 DESC, 3 DESC;


2->
SELECT DISTINCT JOB
FROM EMP;


3->
SELECT EMPNO AS "Emp#",
       ENAME AS "Employee",
       JOB AS "Job",
       HIREDATE AS "Hire Date"
FROM EMP;


4->
SELECT ENAME || ', ' || JOB AS "Employee and Title"
FROM EMP;


5->
SELECT ENAME || ', ' || JOB || ', ' || HIREDATE || ', ' || MGR AS THE_OUTPUT
FROM EMP;

6->
SELECT ENAME, JOB, HIREDATE
FROM EMP
WHERE ENAME IN ('SCOTT', 'TURNER')
ORDER BY HIREDATE ASC;


7->
SELECT ENAME, DEPTNO
FROM EMP
WHERE DEPTNO IN (20, 30)
ORDER BY ENAME ASC;


8->
SELECT ENAME AS "Employee",
       SAL AS "Monthly Salary"
FROM EMP
WHERE SAL BETWEEN 2000 AND 3000
  AND DEPTNO IN (20, 30)
ORDER BY ENAME ASC;


9->
SELECT ENAME, HIREDATE
FROM EMP
WHERE HIREDATE BETWEEN '01-JAN-81' AND '31-DEC-81';

10->
SELECT ENAME, SAL
FROM EMP
WHERE SAL > &AMOUNT;

11->
SELECT ENAME, JOB
FROM EMP
WHERE MGR IS NULL;

12->
SELECT EMPNO, ENAME, SAL, DEPTNO
FROM EMP
WHERE MGR = &MANAGER_ID
ORDER BY &COLUMN_NAME;

13->
SELECT EMPNO, ENAME, SAL, DEPTNO
FROM EMP
WHERE MGR = &MGR
ORDER BY &COLUMN_NAME;

14->
SELECT ENAME
FROM EMP
WHERE ENAME LIKE '__A%';

15->
SELECT ENAME
FROM EMP
WHERE ENAME LIKE '%A%'
AND ENAME LIKE '%S%';

16->
SELECT ENAME, JOB, SAL
FROM EMP
WHERE JOB = 'CLERK'
AND SAL IN (880, 950, 1300);
